CREATE DATABASE centroDeBuceo;
use centroDeBuceo;

/* CREAR TABLAS & FIJAR CLAVES PRIMARIAS */
CREATE TABLE Clientes (
	cod_Cliente INT AUTO_INCREMENT,
	dni VARCHAR(30) NOT NULL,
	nombre VARCHAR(30) NOT NULL,
	apellidos VARCHAR(30) NOT NULL,
	domicilio VARCHAR(30),
	fechaNacimiento DATE NOT NULL,
	telefono INT,
	email VARCHAR(50),
	licencia VARCHAR(30),
	CONSTRAINT cod_Cliente_PK PRIMARY KEY (cod_Cliente)
);

CREATE TABLE Empleados (
	cod_Empleado INT AUTO_INCREMENT,
	dni VARCHAR(30) NOT NULL,
	nombre VARCHAR(30) NOT NULL,
	apellidos VARCHAR(30) NOT NULL,
	domicilio VARCHAR(30),
	fechaNacimiento DATE NOT NULL,
	telefono INT,
	CONSTRAINT cod_Empleado_PK PRIMARY KEY (cod_Empleado)
);

CREATE TABLE Profesores (
	cod_Profe INT AUTO_INCREMENT,
	cod_Empleado INT,
	licencia VARCHAR(30),
	CONSTRAINT cod_Profe_PK PRIMARY KEY (cod_Profe)
);

CREATE TABLE Secretarios (
	cod_Secretario INT AUTO_INCREMENT,
	cod_Empleado INT,
	CONSTRAINT cod_Secretario_PK PRIMARY KEY (cod_Secretario)
);

CREATE TABLE Vehiculos (
	cod_Vehiculo INT AUTO_INCREMENT,
	marca VARCHAR(30),
	modelo VARCHAR(50),
	matricula VARCHAR(30),
	ultimaInspeccion DATE,
	proximaInspeccion DATE,
	CONSTRAINT cod_Vehiculo_PK PRIMARY KEY (cod_Vehiculo)
);

CREATE TABLE Peticiones (
	cod_Peticion INT AUTO_INCREMENT,
	cod_Cliente INT,
	cod_Actividad INT,
	fecha DATE NOT NULL,
	nr_Inmersiones INT DEFAULT 1,
	nr_Personas INT DEFAULT 1,
	CONSTRAINT cod_Peticion_PK PRIMARY KEY (cod_Peticion)
);

CREATE TABLE Reservas (
	cod_Reserva INT AUTO_INCREMENT,
	cod_Cliente INT,
	cod_Excursion INT,
	nr_Personas INT,
	CONSTRAINT cod_Reserva_PK PRIMARY KEY (cod_Reserva)
);

CREATE TABLE Alquiler_Equipo (
	cod_Alquiler INT AUTO_INCREMENT,
	cod_Reserva INT,
	cod_Equipamiento INT,
	CONSTRAINT cod_Alquiler_PK PRIMARY KEY (cod_Alquiler)
);

CREATE TABLE Grupos_Excursiones (
	cod_Excursion INT AUTO_INCREMENT,
	cod_Actividad INT,
	cod_Sitio INT,
	cod_Empleado INT,
	cod_Vehiculo INT,
	fecha DATE NOT NULL,
	hora TIME NOT NULL,
	CONSTRAINT cod_Excursion_PK PRIMARY KEY (cod_Excursion)
);

CREATE TABLE Actividades (
	cod_Actividad INT AUTO_INCREMENT,
	denominacion VARCHAR(50),
	precioPorPersona DOUBLE,
	CONSTRAINT cod_Actividad_PK PRIMARY KEY (cod_Actividad)
);

CREATE TABLE Sitios_Buceo (
	cod_Sitio INT AUTO_INCREMENT,
	nombre VARCHAR(50),
	localizacion VARCHAR(100),
	barco BOOLEAN DEFAULT FALSE,
	CONSTRAINT cod_Sitio_PK PRIMARY KEY (cod_Sitio)
);

CREATE TABLE Equipamiento (
	cod_Equipamiento INT AUTO_INCREMENT,
	cod_Proveedor INT,
	nombre VARCHAR(50),
	precioInmersion DOUBLE,
	ultimaInspeccion DATE,
	proximaInspeccion DATE,
	alquilado BOOLEAN,
	CONSTRAINT cod_Equipamiento_PK PRIMARY KEY (cod_Equipamiento)
);

CREATE TABLE Proveedores (
	cod_Proveedor INT AUTO_INCREMENT,
	nombre VARCHAR(30),
	pais VARCHAR(30),
	telefono INT,
	email VARCHAR(30),
	CONSTRAINT cod_Proveedor_PK PRIMARY KEY (cod_Proveedor)
);

CREATE TABLE Facturas (
	cod_Factura INT AUTO_INCREMENT,
	cod_Reserva INT,
	precio DOUBLE,
	pagado BOOLEAN DEFAULT false,
	CONSTRAINT cod_Factura_PK PRIMARY KEY (cod_Factura)
);

	/* GRUPOS_EXCURSIONES: Añadir FK cod_Empleado */
